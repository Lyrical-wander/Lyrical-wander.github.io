class QLearningAgent:
    # ÂàùÂßãÂåñÔºåÂèÇÊï∞ÔºöÁä∂ÊÄÅÊï∞ÔºåÂä®‰ΩúÊï∞ÔºåÂ≠¶‰π†ÁéáÔºåÊäòÊâ£Âõ†Â≠êÔºàÂ•ñÂä±Ë°∞ÂáèÔºâÔºåË¥™Â©™Â∫¶
    def __init__(self, n_states, n_actions, learning_rate=0.1, gamma=0.99, epsilon=0.1):
        self.q_table = np.zeros((n_states, n_actions))      #ÂàùÂßãÂåñQË°®ÔºåQË°®ÁöÑÂàùÂßãÂåñ‰∏∫0Ôºå‰πüÂèØ‰ª•ÂàùÂßãÂåñ‰∏∫ÈöèÊú∫ÂÄº
        self.learning_rate = learning_rate                  #ÂàùÂßãÂåñÂ≠¶‰π†ÁéáÔºåÂ≠¶‰π†ÁéáÈªòËÆ§‰∏∫0.1
        self.discount_factor = gamma                        #ÂàùÂßãÂåñÊäòÊâ£Âõ†Â≠êÔºåÊäòÊâ£Âõ†Â≠êÈªòËÆ§‰∏∫0.99
        self.epsilon = epsilon                              #ÂàùÂßãÂåñË¥™Â©™Â∫¶ÔºåË¥™Â©™Â∫¶ÈªòËÆ§‰∏∫0.1
        # ÊâìÂç∞ÂàùÂßãÂåñQË°®

    def choose_action(self, state):
        # Âà§Êñ≠Ë¥™Â©™Â∫¶
        if np.random.uniform(0, 1) < self.epsilon:
            # Â¶ÇÊûúËøõÂÖ•‰∫ÜË¥™Â©™Áä∂ÊÄÅÔºåÈÇ£‰πàagentÂ∞Ü‰ºöÂ∞ùËØïÈöèÊú∫ÈÄâÊã©‰∏Ä‰∏™Âä®‰ΩúÔºåÊ≠§Ê≠•Áß∞‰πã‰∏∫„ÄêÊé¢Á¥¢„Äë
            return np.random.choice(self.q_table.shape[1])
        else:
            # ÈÄâÊã©ÊúÄ‰ºòÂä®‰ΩúÔºåÂç≥Âú®QË°®‰∏≠ËØ•Áä∂ÊÄÅ‰∏ãÊúÄÂ§ßÁöÑQÂÄºÔºåÊ≠§Ê≠•Áß∞‰πã‰∏∫„ÄêÂà©Áî®„Äë

    def update(self, state, action, reward, next_state, done):
        best_next_action = np.argmax(self.q_table[next_state])

        td_error = td_target - self.q_table[state][action]
        self.q_table[state][action] += self.learning_rate * td_error
